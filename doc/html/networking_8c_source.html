19 int init_networking(char* remote_host)
20 {
21  /* Init the libraries that we use */
22  if (SDL_Init(0) == -1) {
23  fprintf(stderr, "SDL_Init: %s\n", SDL_GetError());
24  return 1;
25  }
26  if (SDLNet_Init() == -1) {
27  fprintf(stderr, "SDLNet_Init: %s\n", SDLNet_GetError());
28  return 1;
29  }
30 
31  // Resolve the host name of the server
32  if (SDLNet_ResolveHost(&ipaddress, remote_host, MORGENLICHTD_PORT) == -1)
33  {
34  fprintf(stderr, "SDLNet_ResolveHost: %s\n", SDLNet_GetError());
35  return 1;
36  }
37 
38  // Connect to the server
39  if ((server_sock = SDLNet_TCP_Open(&ipaddress)) == NULL) {
40  fprintf(stderr, "SDLNet_TCP_Open: %s\n", SDLNet_GetError());
41  return 1;
42  }
43 
44  return 0;
45 }

47 int network_write(char* message)
48 {
49  int len = strlen(message) + 1; // add one for the terminating NULL
50  int sent = SDLNet_TCP_Send(server_sock, message, len);
51 
52  if (sent < len) {
53  fprintf(stderr, "Could not write to the server: %s\n", SDLNet_GetError());
54  // It may be good to disconnect because it is likely invalid now.
55  return 1;
56  }
57 
58  return 0;
59 }

61 int network_read(int maxlen, char* buff)
62 {
63  if (SDLNet_TCP_Recv(server_sock, buff, maxlen) <= 0) {
64  // An error may have occured, but sometimes you can just ignore it
65  // It may be good to disconnect sock because it is likely invalid now.
66  fprintf(stderr, "Could not read from the server: %s\n", SDLNet_GetError());
67  return 1;
68  }
69 
70  return 0;
71 }
